# COMPLEX ENCRYPTION FUNCTION
def encrypt(password):
    result = []
    for i, digit in enumerate(password):
        d = int(digit)
        
        # Layer 1: Base transformation
        value = d * 13
        value += (i + 1) * 17
        
        # Layer 2: Position-based conditional
        if i % 3 == 0:
            value ^= 0x2A
            value += 50
        elif i % 3 == 1:
            value = (value * 3) % 256
            value += 75
        else:
            value = ((value << 2) & 0xFF) | (value >> 6)
            value += 100
        
        # Layer 3: Prime adjustment
        primes = [2, 3, 5, 7, 11, 13, 17, 19]
        value += primes[i] * 5
        
        # Layer 4: Final encoding
        value = (value * 2) + 128
        if value > 999:
            value = (value % 900) + 100
        
        result.append(value)
    
    return result
